[docs]    def from_sympy(self, a):
        """Convert SymPy's Integer to `dtype`. """
        if a.is_Rational:
            return GMPYRational(a.p, a.q)
        elif a.is_Float:
            from ..domains import RR
            return GMPYRational(*RR.to_rational(a))
        else:
            raise CoercionFailed("expected `Rational` object, got %s" % a)
